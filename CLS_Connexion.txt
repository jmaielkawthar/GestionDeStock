class CLS_Connexion
{
  public bool ConnexionValide(DbStockContext db, string Nom, string Mot_De_Passe )
  {
	  Utilisateur U= new Utilisateur(); // table utilisateur
	  U.NomUtilisateur = Nom;
	  U.Mot_De_Passe = Mot_De_Passe;
	  if(db.Utilisateurs.SingleOrDefault(s=> s.NomUtilisateur == Nom && s.Mot_De_Passe== Mot_De_Passe)!=null) // si le nom d'user et le mot de passe existe dan la base de donn√©e
	  {
	  return true;
	  }
	  else // si n'existe pas
	  {
	   return false;
	  }
  }
}
